# CMakeLists for Base directory
PROJECT(FaceNix)
SET(HEADER_FILE ../../include)
INCLUDE_DIRECTORIES(${HEADER_FILE} )
SET(Base_SRCS fxBase.c )
FIND_PACKAGE( OpenCV REQUIRED )
# shared library
ADD_LIBRARY(Base SHARED ${Base_SRCS})

# static library
ADD_LIBRARY(Base STATIC ${Base_SRCS})
TARGET_LINK_LIBRARIES(Base Error ${OpenCV_LIBS})
SET_TARGET_PROPERTIES(Base PROPERTIES OUTPUT_NAME "Base")
SET_TARGET_PROPERTIES(Base PROPERTIES CLEAN_DIRECT_OUTPUT 1)
