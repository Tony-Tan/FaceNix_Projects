

# CMakeLists for Base directory
PROJECT(FaceNix)
SET(HEADER_FILE ../../include)
SET(ANN_SRCS fxBPNerveNetwork.c
${HEADER_FILE}/fxBPNerveNetwork.h
)
FIND_PACKAGE(OpenMP REQUIRED)
if(OPENMP_FOUND)
message("OPENMP FOUND")
set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} ${OpenMP_C_FLAGS}")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${OpenMP_CXX_FLAGS}")
set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} ${OpenMP_EXE_LINKER_FLAGS}")
endif()
# shared library
ADD_LIBRARY(ANN SHARED ${ANN_SRCS})

# static library
ADD_LIBRARY(ANN STATIC ${ANN_SRCS})
TARGET_LINK_LIBRARIES(ANN Error Base)
SET_TARGET_PROPERTIES(ANN PROPERTIES OUTPUT_NAME "ANN")
SET_TARGET_PROPERTIES(ANN PROPERTIES CLEAN_DIRECT_OUTPUT 1)
