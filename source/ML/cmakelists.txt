# CMakeLists for ML directory
PROJECT(FaceNix)

${HEADER_FILE}/AdaBoost.h 
${HEADER_FILE}/CasCade.h
)

SET_TARGET_PROPERTIES(ML PROPERTIES CLEAN_DIRECT_OUTPUT 1)
